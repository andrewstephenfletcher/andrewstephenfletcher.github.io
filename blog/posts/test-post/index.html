Testing Interactive Plots

To verify my AI Safety visualization pipeline, here is a simple sine wave generated using Matplotlib.

import matplotlib.pyplot as plt
import numpy as np

# Data for the sine wave
x = np.linspace(0, 2 * np.pi, 100)
y = np.sin(x)

# Create the plot
plt.figure(figsize=(8, 4))
plt.plot(x, y, label='Sine Wave', color='blue', linewidth=2)
plt.title("Baseline Visualization Test")
plt.xlabel("Radians")
plt.ylabel("Amplitude")
plt.grid(True, linestyle='--', alpha=0.7)
plt.legend()
plt.show()

This basic plotting now we try an interactive plot.

import plotly.graph_objects as go
import numpy as np

# Create the signal data
x = np.linspace(0, 10, 500)
freqs = np.arange(1, 6, 1)

fig = go.Figure()

# Add a trace for each frequency (hidden by default except the first)
for f in freqs:
    fig.add_trace(
        go.Scatter(
            visible=False,
            line=dict(color="#00CED1", width=3),
            name=f"f = {f}",
            x=x,
            y=np.sin(f * x)
        )
    )

# Make the first trace visible
fig.data[0].visible = True

# Create the slider steps
steps = []
for i in range(len(fig.data)):
    step = dict(
        method="update",
        args=[{"visible": [False] * len(fig.data)},
              {"title": f"Sine Wave Frequency: {i+1}Hz"}],
        label=str(i+1)
    )
    step["args"][0]["visible"][i] = True
    steps.append(step)

# Add the slider to the layout
sliders = [dict(
    active=0,
    currentvalue={"prefix": "Frequency: "},
    pad={"t": 50},
    steps=steps
)]

fig.update_layout(sliders=sliders, title="Interactive Frequency Test")
fig.show()

Now a correlation matrix plot.

import plotly.express as px
import numpy as np
import pandas as pd

# Generate dummy AI Safety metrics data
data = {
    'Reward': [0.9, 0.8, 0.2, 0.4, 0.1],
    'Safety_Buffer': [0.1, 0.2, 0.8, 0.6, 0.9],
    'Compute_Used': [1.0, 0.8, 0.5, 0.3, 0.2],
    'Alignment_Score': [0.8, 0.7, 0.3, 0.5, 0.2]
}
df = pd.DataFrame(data)
corr = df.corr()

# Create the heatmap
fig = px.imshow(
    corr,
    text_auto=".2f", # Shows the correlation values inside the cells
    aspect="auto",
    color_continuous_scale='RdBu_r', # Red/Blue diverging scale
    zmin=-1, zmax=1,                 # Standard correlation limits
    title="Feature Correlation Matrix: Safety Metrics"
)

fig.update_layout(
    plot_bgcolor='white',
    font=dict(family="Inter, sans-serif", size=12) # Matches common Quarto themes
)

fig.show()
